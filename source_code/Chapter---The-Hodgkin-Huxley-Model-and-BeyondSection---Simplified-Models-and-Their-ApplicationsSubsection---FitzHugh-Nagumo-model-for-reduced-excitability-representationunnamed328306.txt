import numpy as np
import matplotlib.pyplot as plt

# Parameters
epsilon = 0.08
a = 0.7
b = 0.8
I_ext = 0.5
dt = 0.01
t_max = 100
steps = int(t_max / dt)

# Initial conditions
v = 0.0
w = 0.0

# Arrays to store results
v_values = []
w_values = []
time = []

# Euler method integration
for i in range(steps):
    dv = (v - (v**3)/3 - w + I_ext) * dt
    dw = epsilon * (v + a - b * w) * dt
    v += dv
    w += dw
    v_values.append(v)
    w_values.append(w)
    time.append(i * dt)

# Plot results
plt.plot(time, v_values, label='v (Membrane Potential)')
plt.plot(time, w_values, label='w (Recovery Variable)')
plt.xlabel('Time')
plt.ylabel('Value')
plt.legend()
plt.show()
