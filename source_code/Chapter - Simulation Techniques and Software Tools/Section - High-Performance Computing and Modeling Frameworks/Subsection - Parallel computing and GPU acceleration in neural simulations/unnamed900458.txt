from neuron import h, gui

# Create neurons
soma = h.Section(name='soma')
soma.insert('hh')

# Define synaptic connections
syn = h.ExpSyn(soma(0.5))
syn.e = 0  # reversal potential
syn.tau = 2  # time constant

# Define stimulation
stim = h.IClamp(soma(0.5))
stim.delay = 5
stim.dur = 1
stim.amp = 0.1

# Run simulation
h.tstop = 20
h.run()
