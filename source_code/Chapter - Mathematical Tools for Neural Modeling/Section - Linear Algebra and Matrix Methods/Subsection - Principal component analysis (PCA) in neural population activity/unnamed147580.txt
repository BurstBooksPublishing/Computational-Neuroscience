import numpy as np
from sklearn.decomposition import PCA

# Example neural data: 10 neurons, 100 time points
neural_data = np.random.rand(10, 100)

# Perform PCA
pca = PCA(n_components=2)  # Reduce to 2 dimensions
transformed_data = pca.fit_transform(neural_data.T)  # Transpose to match PCA input format

# Explained variance
explained_variance = pca.explained_variance_ratio_
