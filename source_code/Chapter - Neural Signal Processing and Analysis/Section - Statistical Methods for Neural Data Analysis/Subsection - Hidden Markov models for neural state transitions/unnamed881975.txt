from hmmlearn import hmm
import numpy as np

# Example neural data (spike counts)
neural_data = np.array([[0, 1, 2, 1, 0, 1, 2, 3, 2, 1]]).T

# Define the HMM with 3 hidden states
model = hmm.GaussianHMM(n_components=3, covariance_type="diag", n_iter=100)

# Fit the model to the neural data
model.fit(neural_data)

# Predict the most likely sequence of hidden states
hidden_states = model.predict(neural_data)

print("Hidden states:", hidden_states)
