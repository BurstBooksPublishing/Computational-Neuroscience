import numpy as np

def global_workspace_simulation(inputs, weights):
    integrated_info = np.dot(weights, inputs)
    return integrated_info

inputs = np.array([0.5, 0.7, 0.3])
weights = np.array([0.4, 0.6, 0.2])
result = global_workspace_simulation(inputs, weights)
print("Integrated Information:", result)
