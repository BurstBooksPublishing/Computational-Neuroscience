from multiprocessing import Pool

def simulate_neuron(tau_m, R_m, I_syn, t):
    V = np.zeros_like(t)
    I_syn_vec = np.array([I_syn(ti) for ti in t])
    dV_dt = (-V + R_m * I_syn_vec) / tau_m
    return np.cumsum(dV_dt) * dt

# Parallel simulation
with Pool() as pool:
    results = pool.starmap(simulate_neuron, [(tau_m, R_m, I_syn, t) for _ in range(1000)])
