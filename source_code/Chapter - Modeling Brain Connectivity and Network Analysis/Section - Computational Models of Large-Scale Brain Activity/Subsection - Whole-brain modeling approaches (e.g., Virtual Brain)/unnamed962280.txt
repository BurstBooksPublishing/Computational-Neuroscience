import numpy as np
import matplotlib.pyplot as plt

# Parameters
c_EE = 12
c_EI = 4
c_IE = 13
c_II = 11
P = 1
Q = 1
tau_E = 1
tau_I = 1
dt = 0.01
T = 100
steps = int(T / dt)

# Initial conditions
E = np.zeros(steps)
I = np.zeros(steps)
E[0] = 0.1
I[0] = 0.1

# Wilson-Cowan model
for t in range(1, steps):
    dE = (-E[t-1] + sigmoid(c_EE * E[t-1] - c_IE * I[t-1] + P)) / tau_E
    dI = (-I[t-1] + sigmoid(c_EI * E[t-1] - c_II * I[t-1] + Q)) / tau_I
    E[t] = E[t-1] + dE * dt
    I[t] = I[t-1] + dI * dt

# Plot results
plt.plot(np.arange(0, T, dt), E, label='Excitatory')
plt.plot(np.arange(0, T, dt), I, label='Inhibitory')
plt.xlabel('Time')
plt.ylabel('Activity')
plt.legend()
plt.show()
