def retinotopic_map(input_space, learning_rate=0.1, epochs=100):
    # Initialize weights
    weights = initialize_weights(input_space)
    for epoch in range(epochs):
        for input_point in input_space:
            # Find the best matching unit (BMU)
            bmu = find_bmu(input_point, weights)
            # Update weights
            weights = update_weights(weights, bmu, input_point, learning_rate)
    return weights
