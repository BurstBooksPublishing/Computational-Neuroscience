import numpy as np

def predictive_coding(x, x_hat, learning_rate=0.01):
    # Compute prediction error
    e = x - x_hat
    # Update prediction
    x_hat += learning_rate * e
    return x_hat, e

# Example usage
x = np.array([1.0, 2.0, 3.0])  # Observed input
x_hat = np.array([0.5, 1.5, 2.5])  # Predicted input
x_hat, e = predictive_coding(x, x_hat)
print("Updated prediction:", x_hat)
print("Prediction error:", e)
