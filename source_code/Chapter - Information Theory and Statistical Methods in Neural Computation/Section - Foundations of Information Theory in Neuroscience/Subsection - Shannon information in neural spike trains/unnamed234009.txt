import numpy as np
from scipy.stats import entropy

def shannon_information(spike_train):
    # Discretize spike train into binary sequence
    binary_spikes = (spike_train > 0).astype(int)
    
    # Compute probabilities of spike (1) and no spike (0)
    p_spike = np.mean(binary_spikes)
    p_no_spike = 1 - p_spike
    
    # Compute Shannon entropy
    return entropy([p_spike, p_no_spike], base=2)
